{
    Console.Write(message);

}

string[] FillArray(int lenght)
{
    string[] array = new string[lenght];
    for (int i = 0; i < lenght; i++)
    {
        array[i] = PrintString($"Введите строку массива {i + 1}: ");
    }
    return array;
}
